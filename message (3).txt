    function renderTop() {
        const allRedux = [];
        familyBrands.forEach(brand => {
            allRedux.push(brand.current);
            brand.versions.forEach(v => allRedux.push(v));
        });
        const top3 = [...allRedux].sort((a, b) => b.downloads - a.downloads).slice(0, 3);
        
        const container = document.getElementById('topReduxList');
        container.innerHTML = top3.map((item, idx) => {
            const medal = idx === 0 ? '🥇' : idx === 1 ? '🥈' : '🥉';
            return `
                <div class="top-item">
                    <div class="top-buttons">
                        <button class="top-review-btn ${getButtonClass(item.hasLinks)}" data-name="${item.name.replace(/'/g, "\\'")}" data-url="${item.reviewUrl}" ${!item.hasLinks ? 'disabled' : ''}>🔍</button>
                        <button class="top-download-btn ${getButtonClass(item.hasLinks)}" data-name="${item.name.replace(/'/g, "\\'")}" data-url="${item.downloadUrl}" ${!item.hasLinks ? 'disabled' : ''}>⬇️</button>
                        <button class="top-copy-btn" data-name="${item.name.replace(/'/g, "\\'")}" data-id="${item.id}">🔗</button>
                    </div>
                    <div class="top-rank">${medal}</div>
                    <div class="top-info">
                        <div class="top-name">${item.name}</div>
                        <div class="top-date">📅 ${item.date}</div>
                    </div>
                </div>
            `;
        }).join('');
        
        document.querySelectorAll('.top-review-btn.active-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                e.stopPropagation();
                openReview(btn.getAttribute('data-url'), btn.getAttribute('data-name'), e);
            });
        });
        document.querySelectorAll('.top-download-btn.active-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                e.stopPropagation();
                handleDownload(btn.getAttribute('data-url'), btn.getAttribute('data-name'), e);
            });
        });
        document.querySelectorAll('.top-copy-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                e.stopPropagation();
                copyModLink(btn.getAttribute('data-id'), btn.getAttribute('data-name'));
            });
        });
    }

    function renderBrands() {
        const container = document.getElementById('brandsList');
        let filtered = [...familyBrands];
        
        if (familySearchQuery) {
            filtered = filtered.filter(brand => 
                brand.name.toLowerCase().includes(familySearchQuery.toLowerCase()) ||
                brand.current.name.toLowerCase().includes(familySearchQuery.toLowerCase()) ||
                brand.versions.some(v => v.name.toLowerCase().includes(familySearchQuery.toLowerCase()))
            );
        }
        
        if (currentFilter !== 'all' && currentFilter !== 'family') {
            filtered = [];
        }
        
        const statsSpan = document.getElementById('searchStats');
        if (statsSpan) {
            statsSpan.textContent = familySearchQuery ? `🔍 найдено: ${filtered.length} из ${familyBrands.length}` : '';
        }
        
        if (filtered.length === 0 && currentFilter !== 'all' && currentFilter !== 'family') {
            container.innerHTML = '';
            return;
        }
        
        if (filtered.length === 0) {
            container.innerHTML = '<div class="no-results">🔍 Ничего не найдено</div>';
            return;
        }
        
        container.innerHTML = filtered.map(brand => `
            <div class="family-brand-item" id="mod-${brand.current.id}">
                <div class="brand-main-row">
                    <div class="brand-buttons">
                        <button class="btn-review ${getButtonClass(brand.current.hasLinks)}" data-name="${brand.current.name.replace(/'/g, "\\'")}" data-url="${brand.current.reviewUrl}" ${!brand.current.hasLinks ? 'disabled' : ''}>🔍 Обзор</button>
                        <button class="btn-download ${getButtonClass(brand.current.hasLinks)}" data-name="${brand.current.name.replace(/'/g, "\\'")}" data-url="${brand.current.downloadUrl}" ${!brand.current.hasLinks ? 'disabled' : ''}>⬇️ Скачать</button>
                        <button class="btn-copy" data-name="${brand.current.name.replace(/'/g, "\\'")}" data-id="${brand.current.id}">🔗</button>
                    </div>
                    <div class="brand-info">
                        <div class="brand-name">${brand.current.name}</div>
                        <div class="brand-date">📅 ${brand.current.date}</div>
                    </div>
                </div>
                ${brand.versions.length > 0 ? `
                <div class="versions-container">
                    <div>
                        <div class="versions-inner">
                            <div class="versions-label">📜 Старые версии (${brand.name})</div>
                            ${brand.versions.map(v => `
                                <div class="version-item" id="mod-${v.id}">
                                    <div class="version-buttons">
                                        <button class="version-review-btn ${getButtonClass(v.hasLinks)}" data-name="${v.name.replace(/'/g, "\\'")}" data-url="${v.reviewUrl}" ${!v.hasLinks ? 'disabled' : ''}>🔍</button>
                                        <button class="version-download-btn ${getButtonClass(v.hasLinks)}" data-name="${v.name.replace(/'/g, "\\'")}" data-url="${v.downloadUrl}" ${!v.hasLinks ? 'disabled' : ''}>⬇️</button>
                                        <button class="version-copy-btn" data-name="${v.name.replace(/'/g, "\\'")}" data-id="${v.id}">🔗</button>
                                    </div>
                                    <div class="version-info">
                                        <div class="version-name">${v.name}</div>
                                        <div class="version-date">📅 ${v.date}</div>
                                    </div>
                                </div>
                            `).join('')}
                        </div>
                    </div>
                </div>
                ` : ''}
            </div>
        `).join('');
        
        attachFamilyEvents();
    }

    function applyFilters() {
        renderSounds();
        renderGunpac();
        renderTimemap();
        renderBrands();
        renderTop();
    }

    let globalDebounce, familyDebounce;
    const globalSearch = document.getElementById('globalSearch');
    if (globalSearch) {
        globalSearch.addEventListener('input', (e) => {
            clearTimeout(globalDebounce);
            globalDebounce = setTimeout(() => {
                globalSearchQuery = e.target.value;
                applyFilters();
            }, 300);
        });
    }
    
    const familySearch = document.getElementById('brandSearch');
    if (familySearch) {
        familySearch.addEventListener('input', (e) => {
            clearTimeout(familyDebounce);
            familyDebounce = setTimeout(() => {
                familySearchQuery = e.target.value;
                renderBrands();
                renderTop();
            }, 300);
        });
    }

    document.querySelectorAll('.filter-btn').forEach(btn => {
        btn.addEventListener('click', () => {
            document.querySelectorAll('.filter-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            currentFilter = btn.getAttribute('data-filter');
            applyFilters();
        });
    });

    document.querySelectorAll('.sort-btn').forEach(btn => {
        btn.addEventListener('click', () => {
            document.querySelectorAll('.sort-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
            currentSort = btn.getAttribute('data-sort');
            applyFilters();
        });
    });

    const familyHeader = document.getElementById('familyHeader');
    const familyContent = document.getElementById('familyContent');
    if (familyHeader && familyContent) {
        familyHeader.addEventListener('click', () => {
            familyHeader.classList.toggle('open');
            familyContent.classList.toggle('open');
        });
    }

    const themeToggle = document.getElementById('themeToggle');
    const body = document.body;
    const savedTheme = localStorage.getItem('theme');
    if (savedTheme === 'light') body.setAttribute('data-theme', 'light');
    else body.setAttribute('data-theme', 'dark');
    
    if (themeToggle) {
        themeToggle.addEventListener('click', () => {
            const current = body.getAttribute('data-theme');
            if (current === 'light') body.setAttribute('data-theme', 'dark');
            else body.setAttribute('data-theme', 'light');
            localStorage.setItem('theme', body.getAttribute('data-theme'));
            showToast(`🌓 ${body.getAttribute('data-theme') === 'light' ? 'Светлая' : 'Тёмная'} тема`, 800);
        });
    }

    showSkeleton();
    setTimeout(() => {
        renderSounds();
        renderGunpac();
        renderTimemap();
        renderBrands();
        renderTop();
        updateTotalModsCount();
    }, 300);

    document.addEventListener('click', (e) => {
        const toast = document.querySelector('.toast');
        if (toast && !e.target.closest('.toast')) {
            toast.classList.remove('show');
            setTimeout(() => toast.remove(), 300);
        }
    });
